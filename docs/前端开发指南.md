
查看当前布局配置：[src/router/modules/employment.ts](src/router/modules/employment.ts)

## 2. 路由配置

### 2.1 添加新页面到侧边栏

在对应的路由文件中添加子路由即可自动显示在侧边栏。

**示例：在学校端添加"学生管理"页面**

```typescript
// src/router/modules/employment.ts

const schoolRoutes: AppRouteRecord = {
  name: 'School',
  path: '/school',
  component: '/index/index',        // 必须指向布局组件
  meta: {
    title: '学校管理',
    icon: 'ri:government-line',
    roles: ['school_admin', 'system_admin']
  },
  children: [
    {
      path: 'students',              // 路由路径
      name: 'SchoolStudents',        // 路由名称（唯一）
      component: '/school/students', // 页面组件路径
      meta: {
        title: '学生管理',           // 侧边栏显示名称
        keepAlive: true              // 是否缓存页面
      }
    },
    // 新增页面
    {
      path: 'teachers',
      name: 'SchoolTeachers',
      component: '/school/teachers',
      meta: {
        title: '教师管理',
        keepAlive: true
      }
    }
  ]
}
```

### 2.2 路由结构说明

| 属性 | 说明 | 示例 |
|------|------|------|
| `path` | 路由路径 | `'students'`, `'teachers'` |
| `name` | 路由名称 | `'SchoolStudents'` |
| `component` | 组件路径（相对 src/views） | `'/school/students'` |
| `meta.title` | 侧边栏/浏览器标签显示 | `'学生管理'` |
| `meta.keepAlive` | 是否缓存页面 | `true` / `false` |
| `meta.icon` | 菜单图标（可选） | `'ri:user-line'` |
| `meta.roles` | 允许访问的角色 | `['school_admin']` |

## 5. 组合式函数 (Composables)

### 5.1 useTable - 表格数据管理

```typescript
// src/composables/useTable.ts
import { fetchStudentList } from '@/api/student'

export function useStudentTable() {
  const {
    data,
    loading,
    pagination,
    loadData,
    reset
  } = useTable(fetchStudentList)

  const handleSearch = (params: any) => {
    loadData(params)
  }

  const handleDelete = async (row: any) => {
    await ElMessageBox.confirm('确认删除该学生信息？', '提示')
    await deleteStudent(row.id)
    ElMessage.success('删除成功')
    loadData()
  }

  return {
    data,
    loading,
    pagination,
    loadData,
    reset,
    handleSearch,
    handleDelete
  }
}
```

### 5.2 useForm - 表单管理

```typescript
import { useForm } from '@/composables/useForm'

export function useStudentForm() {
  const { formRef, validate, resetFields, clearValidate } = useForm()

  const initialData = {
    name: '',
    gender: '',
    college: '',
    phone: ''
  }

  const formData = reactive({ ...initialData })

  const handleSubmit = async () => {
    const valid = await validate()
    if (!valid) return

    await saveStudent(formData)
    ElMessage.success('保存成功')
  }

  const handleReset = () => {
    Object.assign(formData, initialData)
    resetFields()
  }

  return {
    formRef,
    formData,
    handleSubmit,
    handleReset
  }
}
```

### 6.2 定义 API 接口

```typescript
// src/api/student.ts
import api from '@/utils/http'

interface StudentQuery {
  page?: number
  pageSize?: number
  name?: string
  college?: string
}

export const fetchStudentList = (params: StudentQuery) => {
  return api.get<{ list: Student[]; total: number }>({
    url: '/v1/students',
    params
  })
}

export const fetchStudentById = (id: string) => {
  return api.get<Student>({
    url: `/v1/students/${id}`
  })
}

export const createStudent = (data: Partial<Student>) => {
  return api.post<Student>({
    url: '/v1/students',
    data,
    showSuccessMessage: true
  })
}

export const updateStudent = (id: string, data: Partial<Student>) => {
  return api.put<Student>({
    url: `/v1/students/${id}`,
    data,
    showSuccessMessage: true
  })
}

export const deleteStudent = (id: string) => {
  return api.del<void>({
    url: `/v1/students/${id}`,
    showSuccessMessage: true
  })
}
```

### 6.3 使用 Hooks 方式

### 7.2 定义新的 Store

```typescript
// src/store/modules/student.ts
import { defineStore } from 'pinia'
import { fetchStudentList } from '@/api/student'

interface StudentState {
  students: Student[]
  currentStudent: Student | null
  loading: boolean
}

export const useStudentStore = defineStore('student', {
  state: (): StudentState => ({
    students: [],
    currentStudent: null,
    loading: false
  }),

  actions: {
    async loadStudents(params?: any) {
      this.loading = true
      try {
        const res = await fetchStudentList(params)
        this.students = res.list || []
      } finally {
        this.loading = false
      }
    },

    setCurrentStudent(student: Student | null) {
      this.currentStudent = student
    }
  }
})
```

## 10. 常见问题

### Q: 页面没有显示在侧边栏？

检查：
1. 路由是否添加到正确的父路由下
2. 父路由的 `component` 是否指向 `/index/index`（布局组件）
3. 用户角色是否匹配 `meta.roles`

### Q: 页面没有顶部栏和侧边栏？

确保页面路由的父级 component 指向 `/index/index`，而不是直接指向页面组件。

### Q: API 请求失败？

1. 检查 `src/utils/http/index.ts` 中的 baseURL 配置
2. 检查请求路径是否正确
3. 查看浏览器 Network 面板的具体错误信息

### Q: 如何添加新的图表类型？

在 `src/components/art/echarts/` 目录下创建新的图表组件，参考现有图表实现。
